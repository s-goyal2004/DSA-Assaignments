 // duplicates_check.cpp
 #include <bits/stdc++.h>
 using namespace std;
 bool containsDuplicate(const vector<int>& nums) {
 unordered_set<int> s;
    for (int x : nums) {
        if (s.find(x) != s.end()) return true;
 s.insert(x);
    }
    return false;
 }
 int main() {
    vector<int> nums = {1,2,3,1};
 cout << (containsDuplicate(nums) ? "true\n" : "false\n");
    return 0;
 }
